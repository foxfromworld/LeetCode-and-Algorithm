class Solution(object):
  def shiftingLetters(self, s, shifts):
    """
    :type s: str
    :type shifts: List[int]
    :rtype: str
    """
    s = list(s)
    for i in range(len(shifts)-1, -1, -1):
      s[i] = ord(s[i]) + shifts[i] % 26
      if s[i] > ord('z'):
        s[i] = chr(s[i] - 26)
      else:
        s[i] = chr(s[i])  
      if i - 1 > -1:
          shifts[i-1] += shifts[i]
    return ''.join(s)
