```python
from collections import Counter
import heapq

class Solution(object):
  def topKFrequent(self, nums, k):
    """
    :type nums: List[int]
    :type k: int
    :rtype: List[int]
    """
    count = [[cnt, num] for num, cnt in Counter(nums).items()]
    heap = heapq.nlargest(k, count)
    result = []
    for _, n in heap:
      result.append(n)
    return result
```
