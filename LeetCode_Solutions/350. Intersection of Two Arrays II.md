```python
class Solution:
    def intersect(self, nums1: List[int], nums2: List[int]) -> List[int]:
        result =[]
        if len(nums1) > len(nums2):
            nums1, nums2 = nums2, nums1
        cnt = Counter(nums1)
        for num in nums2:
            if cnt[num] > 0:
                cnt[num] -= 1
                result.append(num)
        return result
```
