```python
from collections import Counter
class Solution(object):
  def leastInterval(self, tasks, n):
    """
    :type tasks: List[str]
    :type n: int
    :rtype: int
    """
    count = Counter(tasks)
    max_val = max([val for val in count.values()])
    extra = list(count.values()).count(max_val)

    return max(len(tasks), (max_val - 1) * (n + 1)+ extra)
```
