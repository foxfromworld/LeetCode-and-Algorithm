class Solution(object):
  def addBinary(self, a, b):
    """
    :type a: str
    :type b: str
    :rtype: str
    """
    if a == '0': return b
    if b == '0': return a
    if len(a) > len(b): b = (len(a)-len(b)) * '0' + b
    elif len(b) > len(a): a = (len(b)-len(a)) * '0' + a
    ret, carry = '', 0
    for i in range(len(a)-1, -1, -1):
      add = int(a[i]) + int(b[i]) + carry
      if add > 2:
        ret = '1' + ret
        carry = 1
      elif add > 1:
        ret = '0' + ret
        carry = 1
      else:
        ret = str(add) + ret
        carry = 0
    return '1' + ret if carry == 1 else ret
