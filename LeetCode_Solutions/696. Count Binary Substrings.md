```python
class Solution(object):
  def countBinarySubstrings(self, s):
    """
    :type s: str
    :rtype: int
    """
    ch, cnt, group = s[0], 1, []
    for i in range(1, len(s)):
      if s[i] == ch:
        cnt += 1
      else:
        group.append(cnt)
        cnt = 1
        ch = s[i]
    group.append(cnt)
    cnt = 0
    for i in range(len(group) - 1):
      cnt += min(group[i], group[i+1])
    return cnt
```
